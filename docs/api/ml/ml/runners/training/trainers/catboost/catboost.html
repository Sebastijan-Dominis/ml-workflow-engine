ml.runners.training.trainers.catboost.catboost

Binary classification training utilities using CatBoost.

This module contains helper routines and a top-level training function used to train binary classification models with CatBoost within the project training framework. It provides deterministic data loading, dynamic import of model-specific pipeline components from the ml.components package, pipeline construction, model definition, and training orchestration.

The public entrypoint is train_binary_classification_with_catboost which returns a fitted sklearn.pipeline.Pipeline combining preprocessing steps and the trained CatBoost model.

 38class CatBoostTrainer(Trainer):
 39    """Concrete trainer for CatBoost models within project training flow."""
 40
 41    def train(
 42        self,
 43        model_cfg: TrainModelConfig,
 44        *,
 45        strict: bool,
 46        failure_management_dir: Path
 47    ) -> TrainOutput:
 48        """Execute end-to-end CatBoost training and return standardized output.
 49
 50        Args:
 51            model_cfg: Validated model configuration for training.
 52            strict: Whether feature/data loading operations should fail strictly.
 53            failure_management_dir: Directory for storing failure-management artifacts.
 54
 55        Returns:
 56            Standardized training output containing model, pipeline, lineage, and metrics.
 57
 58        Raises:
 59            PipelineContractError: Propagated when pipeline/model contract checks
 60                fail in downstream validation helpers.
 61
 62        Notes:
 63            Class weighting and categorical feature handling are resolved
 64            dynamically from configuration and task type before model fitting.
 65
 66        Side Effects:
 67            Performs model training and pipeline fitting, which can be
 68            computationally intensive and may write failure-management artifacts
 69            through downstream helpers.
 70        """
 71
 72        stats: DataStats
 73
 74        X, y, lineage = load_features_and_target(model_cfg, snapshot_selection=None, strict=strict)
 75        splits, splits_info = get_splits(
 76            X=X,
 77            y=y,
 78            split_cfg=model_cfg.split,
 79            data_type=model_cfg.data_type,
 80            task_cfg=model_cfg.task
 81        )
 82
 83        X_train = splits.X_train
 84        y_train = transform_target(
 85            splits.y_train,
 86            transform_config=model_cfg.target.transform,
 87            split_name="train"
 88        )
 89        X_val = splits.X_val
 90        y_val = transform_target(
 91            splits.y_val,
 92            transform_config=model_cfg.target.transform,
 93            split_name="val"
 94        )
 95
 96        input_schema, derived_schema = load_schemas(model_cfg, lineage)
 97
 98        cat_features = get_cat_features(model_cfg, input_schema, derived_schema)
 99
100        pipeline_path = Path(f"{model_cfg.pipeline.path}").resolve()
101        pipeline_cfg_raw = load_yaml(pipeline_path)
102        pipeline_cfg = validate_pipeline_config(pipeline_cfg_raw)
103        pipeline_cfg_hash = compute_model_config_hash(pipeline_cfg.model_dump())
104
105        validate_model_feature_pipeline_contract(
106            model_cfg,
107            pipeline_cfg,
108            cat_features
109        )
110
111        class_weights = {}
112        if model_cfg.task.type == "classification":
113            stats = compute_data_stats(y_train)
114            class_weights = resolve_class_weighting(model_cfg, stats, "catboost")
115
116        model = prepare_model(
117            model_cfg,
118            cat_features=cat_features,
119            class_weights=class_weights,
120            failure_management_dir=failure_management_dir
121        )
122
123        pipeline = build_pipeline_with_model(
124            model_cfg=model_cfg,
125            pipeline_cfg=pipeline_cfg,
126            input_schema=input_schema,
127            derived_schema=derived_schema,
128            model=model
129        )
130
131        model_trained, pipeline_trained = train_catboost_model(
132            model_cfg,
133            steps=pipeline.steps,
134            X_train=X_train,
135            y_train=y_train,
136            X_val=X_val,
137            y_val=y_val
138        )
139
140        metrics = compute_metrics(
141            model=model_trained,
142            pipeline=pipeline_trained,
143            model_cfg=model_cfg,
144            X_train=X_train,
145            y_train=y_train,
146            X_val=X_val,
147            y_val=y_val
148        )
149
150        output = TrainOutput(
151            model=model_trained,
152            pipeline=pipeline_trained,
153            lineage=lineage,
154            metrics=metrics,
155            pipeline_cfg_hash=pipeline_cfg_hash
156        )
157
158        return output
