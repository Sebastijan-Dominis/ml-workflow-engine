ml.config.schemas.model_specs

Canonical Pydantic schemas for model specification configuration.

This module defines shared model-spec structures and validation rules used by both search and training configuration entrypoints.

364class ModelSpecs(BaseModel):
365    """Canonical model specification shared by search and training schemas."""
366
367    problem: str
368    segment: SegmentConfig
369    version: str
370    task: TaskConfig
371    target: TargetConfig
372    segmentation: SegmentationConfig = Field(default_factory=SegmentationConfig)
373    min_rows: int = 10000
374    split: SplitConfig
375    algorithm: AlgorithmConfig
376    model_class: str
377    pipeline: PipelineConfig
378    scoring: ScoringConfig
379    class_weighting: ClassWeightingConfig = Field(default_factory=ClassWeightingConfig)
380    feature_store: FeatureStoreConfig
381    explainability: ExplainabilityConfig = Field(default_factory=ExplainabilityConfig)

383    model_specs_lineage: ModelSpecsLineageConfig
384    meta: MetaConfig = Field(default_factory=MetaConfig, alias="_meta")
385
386    model_config = ConfigDict(extra="forbid", populate_by_name=True)
387
388    @model_validator(mode="after")
389    def validate_task_target_consistency(self):
390        """Enforce consistency between task type and target class metadata.
391
392        Returns:
393            ModelSpecs: Validated model specs instance.
394        """
395
396        # --- Classification rules ---
397        if self.task.type == TaskType.classification:
398            if self.target.classes is None:
399                msg = "Classes must be provided for classification tasks."
400                logger.error(msg)
401                raise ConfigError(msg)
402
403            if self.target.classes.count < 2:
404                msg = (
405                    f"Classes count must be at least 2 for classification tasks, "
406                    f"got {self.target.classes.count}."
407                )
408                logger.error(msg)
409                raise ConfigError(msg)
410
411        # --- Non-classification rules ---
412        else:
413            if self.target.classes is not None:
414                msg = (
415                    f"Classes should not be provided for task type "
416                    f"'{self.task.type}'."
417                )
418                logger.error(msg)
419                raise ConfigError(msg)
420
421        return self
422
423    # Validate that target.transform.enabled is False and target.transform.type is None for non-regression tasks, while for regression tasks, if target.transform.enabled is True, then target.transform.type must be specified
424    @model_validator(mode="after")
425    def validate_target_transform_consistency(self):
426        """Validate target transformation compatibility with task type.
427
428        Args:
429            self: Candidate model specs instance.
430
431        Returns:
432            ModelSpecs: Validated model specs instance."""
433
434        if self.task.type != TaskType.regression:
435            if self.target.transform.enabled:
436                msg = (
437                    f"Target transformation is only applicable for regression tasks. "
438                    f"Found enabled for task type '{self.task.type}'."
439                )
440                logger.error(msg)
441                raise ConfigError(msg)
442            if self.target.transform.type is not None:
443                msg = (
444                    f"Target transformation type should be None for non-regression tasks. "
445                    f"Found '{self.target.transform.type}' for task type '{self.task.type}'."
446                )
447                logger.error(msg)
448                raise ConfigError(msg)
449        else:
450            if self.target.transform.enabled and self.target.transform.type is None:
451                msg = (
452                    "Target transformation type must be specified when target transformation "
453                    "is enabled for regression tasks."
454                )
455                logger.error(msg)
456                raise ConfigError(msg)
457
458        return self
459
460    @model_validator(mode="after")
461    def validate_class_weighting_consistency(self):
462        """Ensure class weighting is only enabled for classification tasks.
463
464        Args:
465            self: Candidate model specs instance.
466
467        Returns:
468            ModelSpecs: Validated model specs instance.
469        """
470
471        if self.task.type != TaskType.classification and self.class_weighting.policy != "off":
472            msg = f"Class weighting is only applicable for classification tasks. Found policy '{self.class_weighting.policy}' for task type '{self.task.type}'."
473            logger.error(msg)
474            raise ConfigError(msg)
475        return self
